114 while (pCurrent->m_pNext)
115 {
116 pCurrent = pCurrent->m_pNext;
117 }
118 XListNode<T>* pTail = new XListNode<T>;
119 pTail->m_pNext = nullptr;
120 pTail->m_T = t;
121 pCurrent->m_pNext = pTail;
122 }
